This page presents the contributions generated by our work group. An extended version of this library [is provided here.](https://www.zotero.org/groups/4390760/ilylabs/library) If you'd like to access the full-text version of one of these works and currently cannot access it, send me an e-mail.

To subscribe to new publications via RSS, use the [Zotero Atom feed](https://api.zotero.org/groups/4390760/items?format=atom&limit=100&sort=date&direction=desc). Note: the site's main RSS feed (`feed.xml`) contains blog posts only — publications are loaded dynamically from Zotero and are not included in it.

<script>
// Bibbase's built-in RSS endpoint is broken (returns empty feed).
// Override the RSS button in the Bibbase hamburger menu to point to the
// working Zotero Atom feed instead.
(function() {
  var zoteroFeed = 'https://api.zotero.org/groups/4390760/items?format=atom&limit=100&sort=date&direction=desc';
  function fixRssLinks() {
    document.querySelectorAll('a[href*="bibbase.org/rss"]').forEach(function(a) {
      a.href = zoteroFeed;
    });
  }
  var observer = new MutationObserver(fixRssLinks);
  observer.observe(document.body, { childList: true, subtree: true });
  setTimeout(fixRssLinks, 5000);
})();
</script>
